How to collect logs from the laptop
1

Open PowerShell as Administrator

Press Win + X → click Windows PowerShell (Admin) or Terminal (Admin)

2

One-time setup (first time only)

Run: Set-ExecutionPolicy -Scope CurrentUser RemoteSigned → press Y

3

Copy & run the script below

Paste into PowerShell → press Enter. Collects event logs + CPU/RAM snapshot. Takes ~30 seconds.

4

Two files saved to Desktop automatically

WinDiag_7Days_*.csv — event logs  |  WinDiag_Perf_*.csv — CPU/RAM data

5

Upload both files in Analyze tab

Come back here → Analyze Logs → upload both CSVs → get full diagnosis + performance report.

WinDiag_7Days.ps1 
# =================================================================
#  WinDiag_7Days.ps1 — Event Logs + CPU/RAM Snapshot (Last 7 Days)
#  Run as Administrator
#  Saves two CSV files to Desktop
# =================================================================

$days    = 7
$start   = (Get-Date).AddDays(-$days)
$stamp   = Get-Date -f "yyyyMMdd_HHmm"
$outLog  = "$env:USERPROFILE\Desktop\WinDiag_7Days_$stamp.csv"
$outPerf = "$env:USERPROFILE\Desktop\WinDiag_Perf_$stamp.csv"
$all     = @()

Write-Host "  WinDiag — Collecting logs + performance data..." -ForegroundColor Cyan
Write-Host "  ------------------------------------------------" -ForegroundColor DarkGray

# ── Event Logs: System / Application / Security ───────────────
foreach ($log in @("System", "Application", "Security")) {
    Write-Host "  Collecting $log log..." -ForegroundColor Gray
    try {
        Get-WinEvent -FilterHashtable @{
            LogName=$log; Level=1,2,3; StartTime=$start
        } -ErrorAction SilentlyContinue | ForEach-Object {
            $all += [PSCustomObject]@{
                Time    = $_.TimeCreated.ToString("yyyy-MM-dd HH:mm:ss")
                Log     = $log
                EventID = $_.Id
                Level   = $_.LevelDisplayName
                Source  = $_.ProviderName
                Message = ($_.Message -replace "`r`n"," " -replace ","," ")
            }
        }
    } catch {}
}

# ── Power / Battery Logs ──────────────────────────────────────
foreach ($pl in @(
    "Microsoft-Windows-Kernel-Power/Operational",
    "Microsoft-Windows-Power-Troubleshooter/Operational"
)) {
    try {
        Get-WinEvent -FilterHashtable @{
            LogName=$pl; StartTime=$start
        } -ErrorAction SilentlyContinue | ForEach-Object {
            $all += [PSCustomObject]@{
                Time    = $_.TimeCreated.ToString("yyyy-MM-dd HH:mm:ss")
                Log     = "Power"
                EventID = $_.Id
                Level   = $_.LevelDisplayName
                Source  = $_.ProviderName
                Message = ($_.Message -replace "`r`n"," " -replace ","," ")
            }
        }
    } catch {}
}

# ── Save Event Log CSV ────────────────────────────────────────
if ($all.Count -gt 0) {
    $all | Sort-Object Time -Descending |
        Export-Csv $outLog -NoTypeInformation -Encoding UTF8
    Write-Host "  Event logs: $($all.Count) events saved" -ForegroundColor Green
} else {
    Write-Host "  No critical events found in last $days days" -ForegroundColor Yellow
}

# ── CPU & RAM Snapshot ────────────────────────────────────────
Write-Host "  Collecting CPU and RAM snapshot..." -ForegroundColor Gray

$os      = Get-CimInstance Win32_OperatingSystem
$cpu     = Get-CimInstance Win32_Processor
$cs      = Get-CimInstance Win32_ComputerSystem
$disk    = Get-CimInstance Win32_LogicalDisk -Filter "DeviceID='C:'"
$battery = Get-CimInstance Win32_Battery -ErrorAction SilentlyContinue

$ramTotalGB  = [math]::Round($os.TotalVisibleMemorySize / 1MB, 1)
$ramFreeGB   = [math]::Round($os.FreePhysicalMemory / 1MB, 1)
$ramUsedGB   = [math]::Round($ramTotalGB - $ramFreeGB, 1)
$ramPct      = [math]::Round(($ramUsedGB / $ramTotalGB) * 100)
$diskFreeGB  = [math]::Round($disk.FreeSpace / 1GB, 1)
$diskTotalGB = [math]::Round($disk.Size / 1GB, 1)
$diskUsedPct = [math]::Round((($diskTotalGB - $diskFreeGB) / $diskTotalGB) * 100)

# Top 10 processes by CPU and RAM
$procCPU = Get-Process | Sort-Object CPU -Descending | Select-Object -First 10
$procRAM = Get-Process | Sort-Object WorkingSet -Descending | Select-Object -First 10

$perf = @()
$perf += [PSCustomObject]@{
    Section="SYSTEM"; Name=$cs.Name; CPU_Model=$cpu.Name
    CPU_Cores=$cpu.NumberOfLogicalProcessors; CPU_Load_Pct=$cpu.LoadPercentage
    RAM_Total_GB=$ramTotalGB; RAM_Used_GB=$ramUsedGB; RAM_Free_GB=$ramFreeGB
    RAM_Used_Pct=$ramPct; Disk_C_Total_GB=$diskTotalGB; Disk_C_Free_GB=$diskFreeGB
    Disk_Used_Pct=$diskUsedPct
    Battery_Pct=if($battery){$battery.EstimatedChargeRemaining}else{"N/A"}
    Collected=$stamp
}
foreach ($p in $procCPU) {
    $perf += [PSCustomObject]@{
        Section="TOP_CPU"; Name=$p.Name; CPU_Model=""; CPU_Cores=""
        CPU_Load_Pct=[math]::Round($p.CPU,1)
        RAM_Total_GB=""; RAM_Used_GB=[math]::Round($p.WorkingSet/1MB,1)
        RAM_Free_GB=""; RAM_Used_Pct=""; Disk_C_Total_GB=""
        Disk_C_Free_GB=""; Disk_Used_Pct=""; Battery_Pct=""; Collected=$stamp
    }
}
foreach ($p in $procRAM) {
    $perf += [PSCustomObject]@{
        Section="TOP_RAM"; Name=$p.Name; CPU_Model=""; CPU_Cores=""
        CPU_Load_Pct=[math]::Round($p.CPU,1)
        RAM_Total_GB=""; RAM_Used_GB=[math]::Round($p.WorkingSet/1MB,1)
        RAM_Free_GB=""; RAM_Used_Pct=""; Disk_C_Total_GB=""
        Disk_C_Free_GB=""; Disk_Used_Pct=""; Battery_Pct=""; Collected=$stamp
    }
}

$perf | Export-Csv $outPerf -NoTypeInformation -Encoding UTF8
Write-Host "  CPU/RAM snapshot saved" -ForegroundColor Green
Write-Host ""
Write-Host "  Done! Files saved to Desktop:" -ForegroundColor Cyan
Write-Host "  1. $outLog"
Write-Host "  2. $outPerf"
Write-Host ""
Invoke-Item "$env:USERPROFILE\Desktop"
Write-Host "  Press any key to close..." -ForegroundColor DarkGray
$null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
