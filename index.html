WinDiag_7Days.ps1 
# ============================================================
#  WinDiag_7Days.ps1  —  Last 7 days of critical Windows logs
#  Run as Administrator
#  Output: WinDiag_7Days_YYYYMMDD_HHMM.csv saved to Desktop
# ============================================================

$days  = 7
$start = (Get-Date).AddDays(-$days)
$out   = "$env:USERPROFILE\Desktop\WinDiag_7Days_$(Get-Date -f yyyyMMdd_HHmm).csv"
$all   = @()

Write-Host ""
Write-Host "  WinDiag — Windows Log Collector (Last $days Days)" -ForegroundColor Cyan
Write-Host "  ---------------------------------------------------" -ForegroundColor DarkGray
Write-Host ""

# --- System, Application, Security logs ---
$logs = @("System", "Application", "Security")
foreach ($log in $logs) {
    Write-Host "  Collecting $log log..." -ForegroundColor Gray
    try {
        Get-WinEvent -FilterHashtable @{
            LogName   = $log
            Level     = 1, 2, 3
            StartTime = $start
        } -ErrorAction SilentlyContinue | ForEach-Object {
            $all += [PSCustomObject]@{
                Time    = $_.TimeCreated.ToString("yyyy-MM-dd HH:mm:ss")
                Log     = $log
                EventID = $_.Id
                Level   = $_.LevelDisplayName
                Source  = $_.ProviderName
                Message = ($_.Message -replace "`r`n", " " -replace ",", " ")
            }
        }
    } catch {}
}

# --- Power & Battery logs ---
$powerLogs = @(
    "Microsoft-Windows-Kernel-Power/Operational",
    "Microsoft-Windows-Power-Troubleshooter/Operational"
)
foreach ($pl in $powerLogs) {
    Write-Host "  Collecting Power log..." -ForegroundColor Gray
    try {
        Get-WinEvent -FilterHashtable @{
            LogName   = $pl
            StartTime = $start
        } -ErrorAction SilentlyContinue | ForEach-Object {
            $all += [PSCustomObject]@{
                Time    = $_.TimeCreated.ToString("yyyy-MM-dd HH:mm:ss")
                Log     = "Power"
                EventID = $_.Id
                Level   = $_.LevelDisplayName
                Source  = $_.ProviderName
                Message = ($_.Message -replace "`r`n", " " -replace ",", " ")
            }
        }
    } catch {}
}

Write-Host ""

# --- Save or report empty ---
if ($all.Count -eq 0) {
    Write-Host "  No critical events found in the last $days days." -ForegroundColor Yellow
} else {
    $all | Sort-Object Time -Descending |
        Export-Csv $out -NoTypeInformation -Encoding UTF8
    Write-Host "  Done! $($all.Count) events collected." -ForegroundColor Green
    Write-Host "  File saved to: $out" -ForegroundColor Cyan
    Write-Host ""
    Invoke-Item (Split-Path $out)
}

Write-Host "  Press any key to close..." -ForegroundColor DarkGray
$null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
